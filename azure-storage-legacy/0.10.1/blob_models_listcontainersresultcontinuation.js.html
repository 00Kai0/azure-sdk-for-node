// Module dependencies.
var azureutil = require('azure-common').util;

/**
* Creates a new ListContainersResultContinuation object.
*/
function ListContainersResultContinuation(blobService, options, nextMarker) {
  if (blobService) {
    this.blobService = blobService;
  }

  if (options) {
    this.options = options;
  }

  if (nextMarker &&
      !azureutil.objectIsEmpty(nextMarker)) {
    this.nextMarker = nextMarker;
  }
}

ListContainersResultContinuation.prototype.getNextPage = function (callback) {
  if (this.nextMarker) {
    var options = this.options ? this.options : {};
    options.marker = this.nextMarker;
    this.blobService.listContainers(options, callback);
  } else {
    callback(new Error('No next page'));
  }
};

ListContainersResultContinuation.prototype.hasNextPage = function () {
  return (this.nextMarker !== undefined && this.nextMarker !== null);
};

module.exports = ListContainersResultContinuation;
